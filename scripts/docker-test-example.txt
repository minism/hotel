# Create a docker network for the stack.
docker network create --driver bridge hotel

# Build the containers
make docker-containers

# Run the master server, exposing the HTTP port but not RPC port.
docker run -d --rm --name master --network hotel -p 3000:3000 hotel_master

# Run the spawner, connecting via the docker network.
docker run -d --rm --name spawner --network hotel -e HOTEL_MASTER_ADDRESS="master:3001" --rm --name spawner --network hotel hotel_spawner
